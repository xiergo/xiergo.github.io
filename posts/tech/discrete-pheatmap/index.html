问题描述#

pheatmap包是R绘图包里的重要一员，在绘制热图方面有着独特的优势，学生信的人基本都用过。可是一般很少用heatmap画离散变量，尤其是画完离散变量后，加个legend非常麻烦。今天我遇到了这个需求：

下图是用pheatmap函数画的图，用的matrix是 Pearson correlation，取值[-1, 1]， 灰色代表NA值，红色越深，约接近1；蓝色越深，越接近0 。

老师又让我画一个离散化的热图。就是用一系列的值将[-1，1]区间截断，每个小区间对应一个颜色。如：颜色设置归档：[0.8,1]，[0.6,0.8)，[0.4,0.6)，[0.2,0.4)，[-1,0.2)。

library(pheatmap)
# x : matrix
# sep.vec : seperator vector, such as c(0.2,0.4,0.6,0.8)
# sep.type: either '[' or ']', if '[', [-1,0.2),[0.2,0.4)......[0.8,1]
# col: color
# min: minimum of x
# max: maximum of x
# ... : extra arguments passed to pheatmap.
pheatmap.discre=function(x,sep.vec,sep.type,col,min,max,...){
  discre=function(x,v,type=']'){
    if(any(duplicated(v))) stop('items in v should be unique')
    if(!type %in% c('[',']')) stop('type can only be "[" or "]"')
    if(is.na(x)) return(NA)
    a=c(x,v)
    a=which(x==sort(a))-1
    if(type==']') a=min(a) else a=max(a)
    return(a)
  }
  
  disc.color=function(col,max,min,sep.vec){
    n=length(col)
    v=c(min,sort(sep.vec),max)
    v1=(v[-1]+v[-length(v)])/2
    v1=(v1-min)/(max-min)
    v1=round(1+(n-1)*v1)
    return(col[v1])
  }
  args=list(...)
  sep.vec=sort(sep.vec)
  x=apply(x,1:2, discre,v=sep.vec,type=sep.type)
  col=disc.color(col,max,min,sep.vec)
  x.max=max(x,na.rm = T)
  x.min=min(x,na.rm = T)
  col=col[(x.min:x.max)+1]
  print(col)
  n=length(col)
  d.legend=(x.max-x.min)/n
  legend_breaks=seq(from=x.min+0.5*d.legend,by=d.legend,length.out = n)
  legend_labels=paste(c(min,sep.vec),'~',c(sep.vec,max))
  legend_labels=legend_labels[(x.min:x.max)+1]
  print(legend_breaks)
  print(legend_labels)
  a=list(x,col,legend_labels,legend_breaks)
  names(a)=c('mat','color','legend_labels','legend_breaks')
  args=c(a,args)
  do.call(pheatmap,args)
}
